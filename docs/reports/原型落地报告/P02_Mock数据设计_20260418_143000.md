## 3. 核心数据结构设计

### 3.1 TranslationJob（翻译作业）

根据后端接口文档5.1节，完整字段如下：

```typescript
interface TranslationJob {
  // 基础信息
  id: number;                           // 作业ID
  source_file_name: string;             // 源文件名
  source_file_path: string;             // 源文件绝对路径
  target_file_path: string | null;      // 输出文件绝对路径
  source_lang: string;                  // 源语言，如 "en"
  target_lang: string;                  // 目标语言，如 "zh"
  source_format: 'docx' | 'pdf';        // 源文件格式
  
  // 关联信息
  session_id: string | null;            // 关联会话ID
  task_run_id: string | null;           // 关联任务运行ID
  current_prompt_bundle_id: number | null; // 当前提示词包ID
  
  // 状态信息
  status: JobStatus;                    // 作业状态
  stage: JobStage;                      // 当前阶段
  
  // 进度信息
  page_count: number;                   // 页面总数
  total_segments: number;               // 总分段数
  completed_segments: number;           // 已完成分段数
  failed_segments: number;              // 失败分段数
  progress_percent: number;             // 总体进度，0-100
  
  // 文档分析
  document_summary: DocumentSummary | null; // 文档分析摘要
  translation_notes: string[];          // 翻译说明数组
  task_prompt_text: string | null;      // 实际使用的主翻译提示词
  
  // 配置选项
  options: JobOptions | null;           // 作业创建选项
  
  // 质量信息
  quality_issue_counts: QualityIssueCounts; // 质量问题计数
  critical_issue_count: number;         // 严重问题数
  warning_issue_count: number;          // 警告问题数
  info_issue_count: number;             // 提示问题数
  export_blocked: boolean;              // 是否禁止导出
  export_status: 'ready' | 'blocked';   // 导出状态
  
  // 产物信息
  artifact_available: boolean;          // 译文文件是否已生成
  
  // 错误信息
  error_message: string | null;         // 错误信息
  
  // 时间信息
  created_at: string;                   // 创建时间（ISO 8601）
  updated_at: string;                   // 更新时间（ISO 8601）
  started_at: string | null;            // 开始时间（ISO 8601）
  completed_at: string | null;          // 完成时间（ISO 8601）
}

// 作业状态枚举
type JobStatus = 
  | 'pending'      // 等待中
  | 'in_progress'  // 进行中
  | 'completed'    // 已完成
  | 'failed'       // 失败
  | 'paused'       // 已暂停
  | 'cancelled';   // 已取消

// 作业阶段枚举
type JobStage = 
  | 'queued'       // 排队中
  | 'analyzing'    // 分析中
  | 'translating'  // 翻译中
  | 'rendering'    // 渲染中
  | 'completed'    // 已完成
  | 'failed';      // 失败

// 文档摘要
interface DocumentSummary {
  title?: string;
  author?: string;
  subject?: string;
  keywords?: string[];
  page_count?: number;
  word_count?: number;
}

// 作业选项
interface JobOptions {
  enable_quality_check?: boolean;
  translation_style?: string;
  glossary_id?: number;
}

// 质量问题计数
interface QualityIssueCounts {
  critical: number;
  warning: number;
  info: number;
}
```

  name: string;
  description: string;
}

## 5. Mock服务实现方案

### 5.1 Mock服务架构

```typescript
// src/services/mockService.ts

class MockService {
  // 获取任务列表
  async getTasks(): Promise<{ jobs: TranslationJob[] }> {
    const response = await fetch('/mock-data/tasks.json');
    return response.json();
  }
  
  // 获取任务详情
  async getTaskDetail(id: number): Promise<{ job: TranslationJob }> {
    const { jobs } = await this.getTasks();
    const job = jobs.find(j => j.id === id);
    if (!job) {
      throw new Error('Task not found');
    }
    return { job };
  }
  
  // 创建任务
  async createTask(data: CreateTaskData): Promise<{ job: TranslationJob }> {
    // 模拟创建任务
    const newJob: TranslationJob = {
      id: Date.now(),
      source_file_name: data.file.name,
      source_file_path: `/uploads/${data.file.name}`,
      target_file_path: null,
      source_lang: data.source_lang,
      target_lang: data.target_lang,
      source_format: data.file.name.endsWith('.pdf') ? 'pdf' : 'docx',
      session_id: null,
      task_run_id: null,
      current_prompt_bundle_id: null,
      status: 'pending',
      stage: 'queued',
      page_count: 0,
      total_segments: 0,
      completed_segments: 0,
      failed_segments: 0,
      progress_percent: 0,
      document_summary: null,
      translation_notes: [],
      task_prompt_text: null,
      options: {
        enable_quality_check: data.enable_quality_check,
        translation_style: data.translation_style,
        glossary_id: data.glossary_id
      },
      quality_issue_counts: { critical: 0, warning: 0, info: 0 },
      critical_issue_count: 0,
      warning_issue_count: 0,
      info_issue_count: 0,
      export_blocked: false,
      export_status: 'ready',
      artifact_available: false,
      error_message: null,
      created_at: new Date().toISOString(),
      updated_at: new Date().toISOString(),
      started_at: null,
      completed_at: null
    };
    
    return { job: newJob };
  }
  
  // 暂停任务
  async pauseTask(id: number): Promise<{ job: TranslationJob }> {
    const { job } = await this.getTaskDetail(id);
    job.status = 'paused';
    job.updated_at = new Date().toISOString();
    return { job };
  }
  
  // 继续任务
  async resumeTask(id: number): Promise<{ job: TranslationJob }> {
    const { job } = await this.getTaskDetail(id);
    job.status = 'in_progress';
    job.updated_at = new Date().toISOString();
    return { job };
  }
  
  // 取消任务
  async cancelTask(id: number): Promise<{ job: TranslationJob }> {
    const { job } = await this.getTaskDetail(id);
    job.status = 'cancelled';
    job.updated_at = new Date().toISOString();
    return { job };
  }
  
  // 删除任务
  async deleteTask(id: number): Promise<{ ok: boolean }> {
    // 模拟删除
    return { ok: true };
  }
  
  // 获取配置
  async getConfig(): Promise<Config> {
    const response = await fetch('/mock-data/config.json');
    return response.json();
  }
}

export const mockService = new MockService();
```

### 8.1 状态转换规则

```
pending → in_progress → completed
                     → failed
                     → paused → in_progress
                     → cancelled
```

### 8.2 阶段转换规则
