**核心代码结构**：
```typescript
// src/pages/TaskList/index.tsx
import { useEffect, useState } from 'react';
import { useTaskStore } from '@/stores/taskStore';
import TaskCard from './TaskCard';
import FilterBar from './FilterBar';
import SearchBar from './SearchBar';
import SortDropdown from './SortDropdown';
import EmptyState from './EmptyState';

export default function TaskList() {
  const { tasks, fetchTasks, loading } = useTaskStore();
  const [filteredTasks, setFilteredTasks] = useState(tasks);
  
  // 筛选状态
  const [statusFilter, setStatusFilter] = useState<string>('all');
  const [langFilter, setLangFilter] = useState<string>('all');
  const [timeFilter, setTimeFilter] = useState<string>('all');
  
  // 排序状态
  const [sortBy, setSortBy] = useState<string>('created_at');
  const [sortOrder, setSortOrder] = useState<'asc' | 'desc'>('desc');
  
  // 搜索状态
  const [searchKeyword, setSearchKeyword] = useState<string>('');
  
  useEffect(() => {
    fetchTasks();
  }, []);
  
  useEffect(() => {
    // 应用筛选、排序、搜索逻辑
    let result = [...tasks];
    
    // 筛选
    if (statusFilter !== 'all') {
      result = result.filter(task => task.status === statusFilter);
    }
    
    // 搜索
    if (searchKeyword) {
      result = result.filter(task => 
        task.source_file_name.toLowerCase().includes(searchKeyword.toLowerCase())
      );
    }
    
    // 排序
    result.sort((a, b) => {
      const aValue = a[sortBy];
      const bValue = b[sortBy];
      return sortOrder === 'asc' 
        ? aValue > bValue ? 1 : -1
        : aValue < bValue ? 1 : -1;
    });
    
    setFilteredTasks(result);
  }, [tasks, statusFilter, langFilter, timeFilter, sortBy, sortOrder, searchKeyword]);
  
  return (
    <div className="task-list-page">
      <div className="header">
        <h1>翻译任务</h1>
        <Button type="primary" onClick={() => navigate('/tasks/new')}>
          创建新任务
        </Button>
      </div>
      
      <div className="toolbar">
        <SearchBar value={searchKeyword} onChange={setSearchKeyword} />
        <FilterBar 
          statusFilter={statusFilter}
          langFilter={langFilter}
          timeFilter={timeFilter}
          onStatusChange={setStatusFilter}
          onLangChange={setLangFilter}
          onTimeChange={setTimeFilter}
        />
        <SortDropdown 
          sortBy={sortBy}
          sortOrder={sortOrder}
          onSortChange={setSortBy}
          onOrderChange={setSortOrder}
        />
      </div>
      
      <div className="task-list">
        {loading ? (
          <Spin />
        ) : filteredTasks.length === 0 ? (
          <EmptyState keyword={searchKeyword} />
        ) : (
          filteredTasks.map(task => (
            <TaskCard key={task.id} task={task} />
          ))
        )}
      </div>
    </div>
  );
}
```

```typescript
// src/pages/TaskCreate/index.tsx
import { useState } from 'react';
import { Steps, Button, message } from 'antd';
import { useNavigate } from 'react-router-dom';
import { useTaskStore } from '@/stores/taskStore';
import UploadStep from './UploadStep';
import ConfigStep from './ConfigStep';
import ConfirmStep from './ConfirmStep';

export default function TaskCreate() {
  const navigate = useNavigate();
  const { createTask } = useTaskStore();
  const [current, setCurrent] = useState(0);
  const [formData, setFormData] = useState({
    file: null,
    source_lang: '',
    target_lang: '',
    glossary_id: null,
    translation_style: 'professional',
    enable_quality_check: true,
  });
  
  const steps = [
    { title: '上传文档', component: UploadStep },
    { title: '配置参数', component: ConfigStep },
    { title: '确认信息', component: ConfirmStep },
  ];
  
  const handleNext = () => {
    setCurrent(current + 1);
  };
  
  const handlePrev = () => {
    setCurrent(current - 1);
  };
  
  const handleSubmit = async () => {
    try {
      const job = await createTask(formData);
      message.success('任务创建成功');
      navigate(`/tasks/${job.id}/translate`);
    } catch (error) {
      message.error('任务创建失败');
    }
  };
  
  const CurrentStepComponent = steps[current].component;
  
  return (
    <div className="task-create-page">
      <h1>创建翻译任务</h1>
      
      <Steps current={current} items={steps} />
      
      <div className="step-content">
        <CurrentStepComponent 
          formData={formData}
          onChange={setFormData}
        />
      </div>
      
      <div className="step-actions">
        {current > 0 && (
          <Button onClick={handlePrev}>上一步</Button>
        )}
        {current < steps.length - 1 && (
          <Button type="primary" onClick={handleNext}>下一步</Button>
        )}
        {current === steps.length - 1 && (
          <Button type="primary" onClick={handleSubmit}>开始翻译</Button>
        )}
      </div>
    </div>
  );
}
```

## 3. 状态管理方案

### 3.1 Zustand Store设计

```typescript
// src/stores/taskStore.ts
import { create } from 'zustand';
import { taskService } from '@/services/taskService';
import type { TranslationJob } from '@/types/task';

interface TaskStore {
  // 状态
  tasks: TranslationJob[];
  currentTask: TranslationJob | null;
  loading: boolean;
  error: string | null;
  
  // 操作
  fetchTasks: () => Promise<void>;
  fetchTaskDetail: (id: number) => Promise<void>;
  createTask: (data: CreateTaskData) => Promise<TranslationJob>;
  updateTask: (id: number, data: Partial<TranslationJob>) => void;
  deleteTask: (id: number) => Promise<void>;
  pauseTask: (id: number) => Promise<void>;
  resumeTask: (id: number) => Promise<void>;
  cancelTask: (id: number) => Promise<void>;
}

export const useTaskStore = create<TaskStore>((set, get) => ({
  tasks: [],
  currentTask: null,
  loading: false,
  error: null,
  
  fetchTasks: async () => {
    set({ loading: true, error: null });
    try {
      const { jobs } = await taskService.getTasks();
      set({ tasks: jobs, loading: false });
    } catch (error) {
      set({ error: error.message, loading: false });
    }
  },
  
  fetchTaskDetail: async (id: number) => {
    set({ loading: true, error: null });
    try {
      const { job } = await taskService.getTaskDetail(id);
      set({ currentTask: job, loading: false });
    } catch (error) {
      set({ error: error.message, loading: false });
    }
  },
  
  createTask: async (data: CreateTaskData) => {
    set({ loading: true, error: null });
    try {
      const { job } = await taskService.createTask(data);
      set(state => ({ 
        tasks: [job, ...state.tasks],
        loading: false 
      }));
      return job;
    } catch (error) {
      set({ error: error.message, loading: false });
      throw error;
    }
  },
  
  updateTask: (id: number, data: Partial<TranslationJob>) => {
    set(state => ({
      tasks: state.tasks.map(task => 
        task.id === id ? { ...task, ...data } : task
      ),
      currentTask: state.currentTask?.id === id 
        ? { ...state.currentTask, ...data }
        : state.currentTask
    }));
  },
  
  deleteTask: async (id: number) => {
    try {
      await taskService.deleteTask(id);
      set(state => ({
        tasks: state.tasks.filter(task => task.id !== id)
      }));
    } catch (error) {
      set({ error: error.message });
      throw error;
    }
  },
  
  pauseTask: async (id: number) => {
    try {
      const { job } = await taskService.pauseTask(id);
      get().updateTask(id, job);
    } catch (error) {
      set({ error: error.message });
      throw error;
    }
  },
  
  resumeTask: async (id: number) => {
    try {
      const { job } = await taskService.resumeTask(id);
      get().updateTask(id, job);
    } catch (error) {
      set({ error: error.message });
      throw error;
    }
  },
  
  cancelTask: async (id: number) => {
    try {
      const { job } = await taskService.cancelTask(id);
      get().updateTask(id, job);
    } catch (error) {
      set({ error: error.message });
      throw error;
    }
  },
}));
```

## 4. 表单验证方案

### 4.1 验证规则定义

```typescript
// src/utils/validation.ts

// 文件验证
export const validateFile = (file: File): string | null => {
  // 格式验证
  const allowedFormats = ['application/pdf', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'];
  if (!allowedFormats.includes(file.type)) {
    return '请上传PDF或DOCX格式的文件';
  }
  
  // 大小验证
  const maxSize = 100 * 1024 * 1024; // 100MB
  if (file.size > maxSize) {
    return '文件大小不能超过100MB';
  }
  
  return null;
};

// 语言验证
export const validateLanguage = (sourceLang: string, targetLang: string): string | null => {
  if (!sourceLang) {
    return '请选择源语言';
  }
  
  if (!targetLang) {
    return '请选择目标语言';
  }
  
  if (sourceLang === targetLang) {
    return '目标语言不能与源语言相同';
  }
  
  return null;
};

// 表单验证
export const validateTaskForm = (formData: CreateTaskData): Record<string, string> => {
  const errors: Record<string, string> = {};
  
  // 文件验证
  if (!formData.file) {
    errors.file = '请上传文件';
  } else {
    const fileError = validateFile(formData.file);
    if (fileError) {
      errors.file = fileError;
    }
  }
  
  // 语言验证
  const langError = validateLanguage(formData.source_lang, formData.target_lang);
  if (langError) {
    errors.language = langError;
  }
  
  return errors;
};
```

### 4.2 实时验证Hook

```typescript
// src/hooks/useFormValidation.ts
import { useState, useCallback } from 'react';
import { debounce } from 'lodash';

export function useFormValidation<T>(
  validateFn: (data: T) => Record<string, string>
) {
  const [errors, setErrors] = useState<Record<string, string>>({});
  
  // 防抖验证
  const debouncedValidate = useCallback(
    debounce((data: T) => {
      const newErrors = validateFn(data);
      setErrors(newErrors);
    }, 300),
    [validateFn]
  );
  
  const validate = (data: T) => {
    debouncedValidate(data);
  };
  
  const clearErrors = () => {
    setErrors({});
  };
  
  const hasErrors = Object.keys(errors).length > 0;
  
  return { errors, validate, clearErrors, hasErrors };
}
```
